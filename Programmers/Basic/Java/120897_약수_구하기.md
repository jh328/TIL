 public int[] solution(int n) {

        List<Integer> list = new ArrayList<>();

            if (n % i == 0) {
                list.add(i);
            }
        }

        int[] answer = new int[list.size()];

        for (int i = 0; i < list.size(); i++) {
            answer[i] = list.get(i);
        }
        return answer;
